- [ ] **Step 1: Add `setGaiaState()` function**

Find `function playSound(type)` (around line 1361) and add BEFORE it:

```js
function setGaiaState(state) {
    const el = document.getElementById('gaia-char');
    if (!el) return;
    el.className = 'gaia-' + state;
    // Swap face parts
    document.getElementById('gaia-eyes-happy').setAttribute('display', state === 'sad' ? 'none' : 'inline');
    document.getElementById('gaia-eyes-sad').setAttribute('display', state === 'sad' ? 'inline' : 'none');
    document.getElementById('gaia-mouth-happy').setAttribute('display', state === 'sad' ? 'none' : 'inline');
    document.getElementById('gaia-mouth-sad').setAttribute('display', state === 'sad' ? 'inline' : 'none');
    // Star burst on happy
    if (state === 'happy') {
        const rect = el.getBoundingClientRect();
        const cx = rect.left + rect.width / 2;
        const cy = rect.top + rect.height / 3;
        const dirs = [[-60,-60],[-80,0],[-60,60],[0,-80],[0,60],[60,-60],[80,0],[60,60]];
        dirs.forEach(([dx, dy]) => {
            const s = document.createElement('div');
            s.className = 'gaia-star-burst';
            s.textContent = '⭐';
            s.style.cssText = `left:${cx}px;top:${cy}px;--dx:${dx}px;--dy:${dy}px`;
            document.body.appendChild(s);
            setTimeout(() => s.remove(), 820);
        });
    }
    // Return to idle after animation
    if (state === 'happy' || state === 'sad') {
        setTimeout(() => {
            el.className = 'gaia-idle';
            document.getElementById('gaia-eyes-happy').setAttribute('display', 'inline');
            document.getElementById('gaia-eyes-sad').setAttribute('display', 'none');
            document.getElementById('gaia-mouth-happy').setAttribute('display', 'inline');
            document.getElementById('gaia-mouth-sad').setAttribute('display', 'none');
        }, state === 'happy' ? 1200 : 900);
    }
}
```
